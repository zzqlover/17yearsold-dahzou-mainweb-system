```go
func main() {
    // 1. 加载 .env 环境变量文件
    envFile := filepath.Join(workDir, ".env")
    // 手动解析 key=value 写入 os.Setenv

    // 2. 初始化配置
    cfg := config.Load()

    // 3. 初始化 JSON 数据库
    database.Initialize(cfg)

    // 4. 创建 Gin 引擎
    r := gin.Default()

    // 5. CORS 中间件（允许的前端地址）
    r.Use(cors.New(cors.Config{
        AllowOrigins: []string{"http://localhost:5173", ...},
        AllowMethods: []string{"GET", "POST", "PUT", "DELETE", "OPTIONS"},
        AllowCredentials: true,
    }))

    // 6. 挂载配置到上下文
    r.Use(func(c *gin.Context) {
        c.Set("config", cfg)
    })

    // 7. 静态文件服务
    r.Static("/uploads", cfg.Uploads)

    // 8. 注册 API 路由
    api := r.Group("/api/v1")
    {
        api.GET("/home", handlers.GetHomeData)
        api.GET("/spots", handlers.GetSpots)
        // ... 更多公开接口
        api.POST("/chat", handlers.ChatWithAssistant)
        api.GET("/weather", handlers.GetWeather)
        api.GET("/captcha", handlers.GetCaptcha)
        api.POST("/verify", handlers.VerifyCaptcha)

        // 管理后台路由（需认证）
        admin := api.Group("/admin")
        admin.Use(authMiddleware) // 认证中间件
        {
            admin.POST("/spots", handlers.CreateSpot)
            // ...
        }
    }

    // 9. 启动服务
    r.Run(":" + cfg.Port) // 默认 5000
}
```

## 三、后端安全机制

### 3.1 滑动验证码（captcha.go）

#### 核心数据结构

```go
type CaptchaStore struct {
    mu    sync.RWMutex
    store map[string]CaptchaInfo  // token → 正确X坐标
}

type CaptchaInfo struct {
    X        int       // 正确的滑块X位置
    CreateAt time.Time // 创建时间（用于过期判断）
}

var captchaStore = CaptchaStore{
    store: make(map[string]CaptchaInfo),
}
```

#### 生成流程（GET /api/v1/captcha）

```
1. generateToken() → 32位随机字符串（a-zA-Z0-9）
2. randomX() → 150~280 之间的随机整数
3. 写入 captchaStore.store[token] = CaptchaInfo{X, CreateAt}
4. 返回 { token, x }
```

#### 验证流程（POST /api/v1/verify）

```
1. 查找 captchaStore.store[token]
   └─ 不存在 → 返回"验证码不存在或已过期"

2. 删除 token（防止重放攻击）

3. abs(offsetX - X) > 5 → 验证失败
   └─ 误差超过5像素视为失败

4. 轨迹校验：
   - len(trajectory) < 5 → 失败（点数太少）
   - 每个点必须有 t（时间戳）和 x（坐标）
   └─ 结构不完整 → 失败
```

#### 防重放机制

- **Token 一次性**：验证后立即从 map 中删除
- **5分钟过期**：后台 goroutine 每分钟清理过期 token
- **轨迹验证**：确保是真实拖动而非直接发包

### 3.2 管理员认证（admin.go）

#### 登录流程

```go
func AdminLogin(c *gin.Context) {
    // 1. 接收 { username, password }
    var input struct {
        Username string `json:"username"`
        Password string `json:"password"`
    }

    // 2. 查找用户（线性遍历 Admins 数组）
    for _, a := range admins {
        if a.Username == input.Username {
            admin = a
            found = true
            break
        }
    }

    // 3. bcrypt 比对密码
    err := bcrypt.CompareHashAndPassword([]byte(admin.PasswordHash), []byte(input.Password))

    // 4. 生成简单 token
    token := admin.Username + "-token-" + time.Now().Format("20060102150405")

    // 5. 返回 token
    c.JSON(200, gin.H{ "token": token, "username": admin.Username })
}
```

#### 密码安全

- 使用 `golang.org/x/crypto/bcrypt` 哈希密码
- 默认 cost = 10，计算耗时约 300ms
- 存储格式：`$2a$10$...`（盐值内嵌在哈希中）
- 验证时使用 `bcrypt.CompareHashAndPassword` 常数时间比较，防止时序攻击

### 4.2 密码加密流程

```javascript
async encryptPassword(password, salt = '') {
  const timestamp = Date.now()
  const randomStr = Math.random().toString(36).substring(2, 15)
  const combined = `${randomStr}${timestamp}${password}${salt}`
  const shaHash = await this.sha256(combined)

  return {
    encrypted: shaHash,     // SHA-256 哈希值
    timestamp,              // 时间戳（用于风控）
    random: randomStr       // 随机字符串（用于盐值）
  }
}
```

**为什么不直接 MD5？**
- MD5 有已知碰撞攻击，且散列强度不足
- SHA-256 是 NIST 推荐的标准，安全强度 256bit
- 配合时间戳+随机盐，即使相同密码每次加密结果也不同

### 4.5 反调试机制（antiDebug）

```javascript
antiDebug: {
  enable() {
    const isDev = location.hostname === 'localhost' || location.hostname === '127.0.0.1'

    this.interval = setInterval(() => {
      const devToolsOpen = securityUtils._checkDevToolsSync()  // 窗口尺寸差
      const debugSlow = securityUtils.detectDebugger()        // 性能计时

      if (devToolsOpen || debugSlow) this.score++

        // 开发环境：立即刷新

            document.body.appendChild(banner)

        }
      }
    }, 500)  // 每500ms检测一次
```

### 4.6 Bot 检测（detectBot）

```javascript
detectBot() {
  if (location.hostname === 'localhost') return false  // 白名单

  let score = 0
  if (navigator.webdriver) score++          // Selenium WebDriver 标志
  if (/HeadlessChrome/.test(navigator.userAgent)) score++  // 无头浏览器
  if (!navigator.plugins || navigator.plugins.length === 0) score++  // 无插件
  if (!navigator.languages || navigator.languages.length === 0) score++  // 无语言
  if (window.chrome && !window.chrome.loadTimes) score++  // Chrome 属性缺失
  if (!window.indexedDB) score++             // 无数据库
  if (!navigator.hardwareConcurrency) score++  // 无CPU核心数
  if (screen.width < 200 || screen.height < 200) score++  // 异常屏幕尺寸

  return score >= 3  // 至少3项异常才判定为 Bot
}
```

### 4.7 行为追踪（behaviorTracker）

```javascript
behaviorTracker: {
  mouseMoves: [],   // [{x, y, t}]
  keyPresses: [],  // [keyCode]
  scrollEvents: [], // [timestamp]
  isHuman: false,

  init() {
    document.addEventListener('mousemove', (e) => {
      this.mouseMoves.push({ x: e.clientX, y: e.clientY, t: Date.now() })
      if (this.mouseMoves.length > 100) this.mouseMoves = this.mouseMoves.slice(-50)
      if (!this.verified) this._analyze()
    }, { passive: true })  // 被动监听，不阻塞滚动

    document.addEventListener('keydown', (e) => {
      this.keyTimestamps.push(Date.now())
      this.keyPresses.push(e.keyCode)
      if (!this.verified) this._analyze()
    })

    window.addEventListener('scroll', () => {
      this.scrollEvents.push(Date.now())
      if (!this.verified) this._analyze()
    })
  },

  _analyze() {
    if (this.mouseMoves.length < 5 || this.keyTimestamps.length < 1) return

    // 1. 非线性移动检测
    let nonLinearMoves = 0
    for (let i = 2; i < this.mouseMoves.length; i++) {
      const angleDiff = 计算三点角度差
      if (angleDiff > 0.1) nonLinearMoves++  // 方向变化
    }

    // 2. 滚动停顿检测
    let scrollPauses = 0
    for (let i = 1; i < this.scrollEvents.length; i++) {
      if (this.scrollEvents[i] - this.scrollEvents[i-1] > 1000) scrollPauses++
    }

    // 3. 键盘间隔检测
    let keyInterval200 = false
    for (let i = 1; i < this.keyTimestamps.length; i++) {
      if (this.keyTimestamps[i] - this.keyTimestamps[i-1] > 200) {
        keyInterval200 = true
        break
      }
    }

    // 综合判定
    if (nonLinearMoves >= 5 && scrollPauses >= 2 && keyInterval200) {
      this.isHuman = true
      this.verified = true
      sessionStorage.setItem('human_verified', 'true')
    }
  },

  // 曲率分析（进阶）
  _curvatureAnalysis() {
    const last30 = this.mouseMoves.slice(-30)
    const speeds = last30.map((m, i) => {
      if (i === 0) return 0
      const dt = m.t - last30[i-1].t
      return Math.sqrt(dx*dx + dy*dy) / dt
    })
    const stddev = 计算标准差
    if (stddev < 0.05) this.isHuman = false  // 速度太均匀 = 机器
  }
}
```

## 十、安全注意事项

### 10.1 前端安全局限

1. **JavaScript 代码可见**：所有前端安全措施可被绕过，不应用于保护高敏感数据
2. **API Key 暴露**：`OPENAI_API_KEY` 等密钥存储在前端 axios 的 baseURL 中，仅混淆而非加密
3. **补充**：敏感操作（如删除数据）应在后端增加额外验证

### 10.2 后端安全要点

1. **JSON 数据库**：无用户级权限隔离，所有管理员共享同一数据库文件
2. **Token 设计**：简单字符串拼接，无签名验证，生产环境应使用 JWT
3. **密码存储**：bcrypt 加盐哈希，安全可靠

### 10.3 建议增强项

- 引入 JWT 替代简单 token
- 添加 API 限流（rate limiting）
- 增加操作审计日志
- 图片上传增加格式校验和病毒扫描
- 数据库增加备份机制
